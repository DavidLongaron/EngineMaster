#version 460
layout(location=0) in vec3 my_vertex_position;
layout(location=1) in vec3 aColor;
layout (location = 2) in vec2 aTexCoord;

out vec3 ourColor;
out vec2 TexCoord;
uniform mat4 proj;
uniform mat4 view;
uniform mat4 model;

void main()
{
gl_Position = vec4(proj*view*model*vec4(my_vertex_position, 1.0));
ourColor= aColor;
TexCoord = vec2(aTexCoord.x, aTexCoord.y);
}
